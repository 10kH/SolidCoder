from functools import lru_cache

sys.setrecursionlimit(200010)

n = int(input())
A = list(map(int, input().split()))

vals = set(A)
vals.add(-1)
sorted_vals = sorted(vals)
rank = {v: i for i, v in enumerate(sorted_vals)}

@lru_cache(None)
def can_win(l: int, r: int, pr: int) -> int:
    if l > r:
        return 0
    pv = sorted_vals[pr]
    # try left
    if l <= r and A[l] > pv:
        npr = rank[A[l]]
        if can_win(l + 1, r, npr) == 0:
            return 1
    # try right
    if l <= r and A[r] > pv:
        npr = rank[A[r]]
        if can_win(l, r - 1, npr) == 0:
            return 1
    return 0

res = can_win(0, n - 1, rank[-1])
print("Alice" if res == 1 else "Bob")

class SegTree:
    def __init__(self, n):
        self.n = n
        self.inf = inf
        self.tree = [inf] * (4 * (n + 1))
        self.lazy = [0] * (4 * (n + 1))

    def _push(self, v, tl, tr):
        if self.lazy[v] != 0:
            self.tree[v] += self.lazy[v]
            if tl != tr:
                self.lazy[2 * v] += self.lazy[v]
                self.lazy[2 * v + 1] += self.lazy[v]
            self.lazy[v] = 0

    def _update_range(self, v, tl, tr, l, r, addend):
        self._push(v, tl, tr)
        if l > r:
            return
        if l <= tl and tr <= r:
            self.lazy[v] += addend
            self._push(v, tl, tr)
            return
        tm = (tl + tr) // 2
        self._update_range(2 * v, tl, tm, l, min(r, tm), addend)
        self._update_range(2 * v + 1, tm + 1, tr, max(l, tm + 1), r, addend)
        self.tree[v] = min(self.tree[2 * v], self.tree[2 * v + 1])

    def range_update(self, l, r, val):
        if l > r:
            return
        self._update_range(1, 1, self.n, l, r, val)

    def _query(self, v, tl, tr, l, r):
        self._push(v, tl, tr)
        if l > r:
            return self.inf
        if l <= tl and tr <= r:
            return self.tree[v]
        tm = (tl + tr) // 2
        left = self._query(2 * v, tl, tm, l, min(r, tm))
        right = self._query(2 * v + 1, tm + 1, tr, max(l, tm + 1), r)
        return min(left, right)

    def query(self, l, r):
        return self._query(1, 1, self.n, l, r)

    def point_set(self, pos, val):
        curr = self.query(pos, pos)
        delta = val - curr
        self.range_update(pos, pos, delta)

        if rank[px] > rank[py]:
            parent[py] = px
