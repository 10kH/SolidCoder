        return 0

    if state[0] > 0:
        new_state = list(state)
        new_state[0] -= 1
        res = max(res, 100 + max_score(tuple(new_state)))
    # single 5 (index 4)
    if state[4] > 0:
        new_state = list(state)
        new_state[4] -= 1
        res = max(res, 50 + max_score(tuple(new_state)))
    # multiples of a kind
    scores = [0, 1000, 200, 300, 400, 500, 600]
    for i in range(6):
        for n in range(3, 7):
            if state[i] >= n:
                new_state = list(state)
                new_state[i] -= n
                pts = (n - 2) * scores[i + 1]
                res = max(res, pts + max_score(tuple(new_state)))
    # straight
    if all(x >= 1 for x in state):
        new_state = list(state)
        for j in range(6):

            new_state[a] -= 2
            new_state[b] -= 2
            new_state[c] -= 2
            res = max(res, 750 + max_score(tuple(new_state)))

def hofstadter_q(n):
    if n == 1 or n == 2:
        return 1
    q = [0] * (n + 1)
    q[1] = 1
    q[2] = 1

def find_longest(s):
    n = len(s)
    max_len = 0
    best_start = float('inf')
    best_sub = ""
    i = 1
    while i < n:
        if (s[i-1] in '!?' and s[i] in '!?' and s[i-1] != s[i]):
            first_char = s[i-1]
            second_char = s[i]
            # find start of first group
            start = i - 1
            while start > 0 and s[start - 1] == first_char:
                start -= 1
            # find end of second group
            end = i
            while end < n and s[end] == second_char:
                end += 1
            length = end - start
            if length > max_len or (length == max_len and start < best_start):
                max_len = length
                best_start = start
                best_sub = s[start:end]
        i += 1
    return best_sub

    return c.isalnum() or c in "'-"

    if not token:
        return ''
    i = len(token) - 1
    suffix = ''
    while i >= 0 and not is_word_char(token[i]):
        suffix = token[i] + suffix
        i -= 1
    word_part = token[:i + 1] if i >= 0 else ''
    length = len(word_part)
    t_count = sum(1 for c in word_part if c.lower() == 't')
    if length > 6 or t_count >= 2:
        new_part = word_part[::-1]
    elif length == 2 or token.endswith(','):
        new_part = word_part.upper()
    elif length == 1:
        new_part = '0'
    else:
        new_part = word_part
    return new_part + suffix

    b = 0

def can_be_increasing(sequence):
    n = len(sequence)
    violations = []
    for i in range(n - 1):
        if sequence[i] >= sequence[i + 1]:
            violations.append(i)
    num_v = len(violations)
    if num_v == 0:
        return True
    if num_v > 1:
        return False
    k = violations[0]
    # Try removing sequence[k]
    can_remove_k = (k == 0) or (sequence[k - 1] < sequence[k + 1])
    # Try removing sequence[k+1]
    can_remove_kp1 = (k + 1 == n - 1) or (sequence[k] < sequence[k + 2])
    return can_remove_k or can_remove_kp1

def calc_tip(p: int, r: int) -> int:
    # Step 1: Round p at the tens place
    rounded_p = ((p + 5) // 10) * 10
    
    # Step 2: Base tip T by dropping the singles place digit
    base_t = rounded_p // 10
    
    # Step 3: Apply rating R
    if r == 1:
        tip = base_t + 1
    elif r == 0:
        tip = base_t - 1
    else:  # r == -1
        tip = (base_t // 2) - 1
    
    # Ensure non-negative
    return max(0, tip)

from collections import deque
import sys

def min_steps(num):
    if num == 0:
        return 0  # Though constraint is 1 <= num
    q = deque([(1, 0)])
    visited = set([1])
    while q:
        cur, steps = q.popleft()
        if cur == num:
            return steps
        for next_val in [cur + 1, cur * 2]:
            if next_val <= num and next_val not in visited:
                visited.add(next_val)
                q.append((next_val, steps + 1))
    return -1  # Unreachable

def almost_everywhere_zero(n_str, k):
    if not n_str or n_str == '0':
        return 0
    digits = [int(c) for c in n_str]
    L = len(digits)
    memo = {}
    
    def dp(pos, tight, started, cnt):
        if pos == L:
            return 1 if started and cnt == k else 0
        key = (pos, tight, started, cnt)
        if key in memo:
            return memo[key]
        ans = 0
        up = digits[pos] if tight else 9
        for d in range(up + 1):
            new_tight = 1 if tight and d == up else 0
            if started == 0 and d == 0:
                new_started = 0
                new_cnt = cnt
            else:
                new_started = 1
                new_cnt = cnt + (1 if d > 0 else 0)
            if new_cnt > k:
                continue
            ans += dp(pos + 1, new_tight, new_started, new_cnt)
        memo[key] = ans
        return ans
    
    return dp(0, 1, 0, 0)

def factor(n):
    if n == 1:
        return {}
    exponents = {}
    # Handle 2 separately
    while n % 2 == 0:
        exponents[2] = exponents.get(2, 0) + 1
        n //= 2
    # Odd factors
    d = 3
    while d * d <= n:
        while n % d == 0:
            exponents[d] = exponents.get(d, 0) + 1
            n //= d
        d += 2
    if n > 1:
        exponents[n] = 1
    return exponents

def spinningRings(innerMax, outerMax):

    m = outerMax + 1
    d = gcd(n, m)
    lcmv = n * (m // d)

    s = min(n, m)

    def update(tt):
        nonlocal ans
        if 0 < tt < ans:
            ans = tt

    # Choose construction with smaller N
    if n <= m:
        # Use outer: N = n, M = m, is_inner = False
        N = n
        M = m
        is_inner = False
    else:
        # Use inner: N = m, M = n, is_inner = True
        N = m
        M = n
        is_inner = True

    p = 0
    while p * M < ans and p < N * 2 + 10:
        if is_inner:
            multiplier = p + 1
            c = (multiplier * n % N)
        else:
            pm_mod = (p * M % N)
            c = (N - pm_mod) % N
        gs = gcd(2, N)
        if c % gs == 0:
            found = False
            if gs == 1:
                inv = pow(2, -1, N)
                bb = (c * inv) % N
                if 1 <= bb < s:
                    if is_inner:
                        tt = p * M + M - bb
                    else:
                        tt = p * M + bb
                    update(tt)
                    found = True

                nh = N // 2
                ch = c // 2
                b1 = ch % nh

                        else:
                            tt = p * M + bb
                        update(tt)
                        found = True
                        break

def to_digit_char(d: int) -> str:
    if 0 <= d <= 9:
        return str(d)
    elif 10 <= d <= 35:
        return chr(ord('A') + d - 10)
    else:
        raise ValueError(f"Invalid digit: {d}")

def converter(num: float, decimals: int = 0, base: float = math.pi) -> str:
    sign = ''
    if num < 0:
        sign = '-'
        num = -num
    if num == 0:
        return '0'

    # Find highest power and initial scale
    power = 0
    scale = 1.0
    while scale * base <= num + 1e-9:
        next_scale = scale * base
        if math.isinf(next_scale):
            break
        scale = next_scale
        power += 1

    # Integer part digits (MSD first)
    current_num = num
    int_digits = []
    for i in range(power, -1, -1):
        digit = math.floor(current_num / scale)
        int_digits.append(int(digit))
        current_num -= digit * scale
        if i > 0:
            scale /= base

    remainder = current_num

    # Fractional part digits
    frac_digits = []
    r = remainder
    for _ in range(decimals):
        r *= base
        digit = math.floor(r)
        frac_digits.append(int(digit))
        r -= digit

    # Build integer part string
    int_str = ''.join(to_digit_char(d) for d in int_digits).lstrip('0')
    if not int_str:
        int_str = '0'

    # Build result
    if decimals == 0:
        result = int_str
    else:
        frac_str = ''.join(to_digit_char(d) for d in frac_digits)
        result = int_str + '.' + frac_str

    return sign + result

def parse_terms(expr):
    terms = []
    i = 0
    n = len(expr)
    while i < n:
        # determine sign
        if i == 0:
            if expr[i] in '+-':
                cursign = 1 if expr[i] == '+' else -1
                i += 1
            else:
                cursign = 1
        else:
            if expr[i] not in '+-':
                i = n
                continue
            cursign = 1 if expr[i] == '+' else -1
            i += 1
        # coefficient
        coeff_str = ''
        while i < n and expr[i].isdigit():
            coeff_str += expr[i]
            i += 1
        coeff = int(coeff_str) if coeff_str else 1
        # variables
        var_list = []
        while i < n and expr[i].islower():
            var_list.append(expr[i])
            i += 1
        key = ''.join(sorted(var_list))
        terms.append((key, cursign * coeff))
    return terms

def simplify(expr):
    if not expr:
        return ''
    terms = parse_terms(expr)
    coeff_dict = defaultdict(int)
    for key, c in terms:
        coeff_dict[key] += c
    monoms = [(k, coeff_dict[k]) for k in coeff_dict if coeff_dict[k] != 0]
    monoms.sort(key=lambda p: (len(p[0]), p[0]))
    terms_str = []
    for key, coeff in monoms:
        abs_c = abs(coeff)
        cstr = str(abs_c) if (abs_c != 1 or not key) else ''
        vstr = key

            terms_str.append('+' + monom)
        else:
            terms_str.append('-' + monom)

def next_smaller(n):
    s = list(str(n))
    # Find the largest i such that s[i] > s[i+1]
    i = len(s) - 2
    while i >= 0:
        if s[i] > s[i + 1]:
            break
        i -= 1
    else:
        # No such i, it's the smallest permutation
        return -1
    
    # Find the largest j > i such that s[j] < s[i]
    j = len(s) - 1
    while j > i:
        if s[j] < s[i]:
            break
        j -= 1
    
    # Swap s[i] and s[j]
    s[i], s[j] = s[j], s[i]
    
    # Reverse the suffix from i+1 to end
    left = i + 1
    right = len(s) - 1
    while left < right:
        s[left], s[right] = s[right], s[left]
        left += 1
        right -= 1
    
    # Check for leading zero
    if s[0] == '0':
        return -1
    
    # Return the integer
    return int(''.join(s))

# Read input and print output

def is_square(x):
    if x < 0:
        return False
    s = int(math.sqrt(x))
    return s * s == x

def sum_of_squares(n):
    # Check if n is a perfect square (1 square)
    if is_square(n):
        return 1
    
    # Check if n is sum of two perfect squares
    s = int(math.sqrt(n)) + 1
    for i in range(1, s):
        if is_square(n - i * i):
            return 2
    
    # Check if n can be expressed as sum of three squares
    # Using Legendre's three-square theorem
    m = n
    while m % 4 == 0:
        m //= 4
    if m % 8 == 7:
        return 4
    else:
        return 3

def proper_fractions(n):
    if n < 2:
        return 0
    result = n
    if n % 2 == 0:
        result -= result // 2
        while n % 2 == 0:
            n //= 2
    p = 3

        if n % p == 0:
            result -= result // p
            while n % p == 0:
                n //= p
        p += 2
    if n > 1:
        result -= result // n
    return result

import sys

sys.setrecursionlimit(10**6)

MOD = 10**9 + 7

class SegTree:
    def __init__(self, size):
        self.n = size + 1
        self.tree = [0] * (4 * self.n)

    def _update(self, node, start, end, idx, val):
        if start == end:
            self.tree[node] = (self.tree[node] + val) % MOD
            return
        mid = (start + end) // 2
        if idx <= mid:
            self._update(2 * node, start, mid, idx, val)
        else:
            self._update(2 * node + 1, mid + 1, end, idx, val)
        self.tree[node] = (self.tree[2 * node] + self.tree[2 * node + 1]) % MOD

    def update(self, idx, val):
        if val == 0:
            return
        self._update(1, 0, self.n - 1, idx, val)

    def _query(self, node, start, end, l, r):
        if r < start or end < l:
            return 0
        if l <= start and end <= r:
            return self.tree[node]
        mid = (start + end) // 2
        left = self._query(2 * node, start, mid, l, r)
        right = self._query(2 * node + 1, mid + 1, end, l, r)
        return (left + right) % MOD

    def query(self, l, r):
        if l > r:
            return 0
        return self._query(1, 0, self.n - 1, l, r)

def get_hmin(suffix, target, m):
    lim = m - 1 - target
    lo = 0
    hi = target
    ans = target + 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if suffix[mid] <= lim:
            ans = mid
            hi = mid - 1
        else:
            lo = mid + 1
    return ans

## Explanation

The solution uses dynamic programming with segment trees to count legal paths while enforcing both horizontal and vertical pushing constraints efficiently.

**Precomputations:**
- `suffix_row[r][c]`: Number of rocks in row `r` from column `c+1` to `m-1`.
- `suffix_col[c][r]`: Number of rocks in column `c` from row `r+1` to `n-1`.
- `hmin[r][k]`: Minimum entry column `L` ≤ `k` such that traversing from `L` to `k` in row `r` is legal, i.e., `m-1 - k ≥ suffix_row[r][L]`, computed via binary search per target `k`.

**Segment Trees:**
- One per column `c` (0 to `m-1`), indexing `s = 0` to `n`, storing ways to exit a row at column `c` with ongoing vertical run having `s` rocks below start row `T`.

**Initialization (Row 0):**
- Entry at column 0.
- Compute `max_exit = m-1 - suffix_row[0][0]`.
- For each valid exit `k ≤ max_exit`, start new vertical run at `T=0`, `s = suffix_col[k][0]`, add 1 way.

**Transitions (Rows 0 to n-2 to next row):**
- Compute `limit = n-1 - (rr+1)` for ending runs at row `rr+1`.
- For each column `c`, `sum_ok[c]` = sum of ways with `s ≤ limit` (query segment tree).
- Prefix sums of `sum_ok` enable O(1) range sums.
- For each target exit `k` in row `rr+1`:
  - `sum_end` = sum of `sum_ok[c]` for `c` from `hmin[rr+1][k]` to `k-1` (ending previous runs).
  - Add `sum_end` to segment tree `k` at `s_new = suffix_col[k][rr+1]` (new runs starting at `rr+1`).
- Continues are implicitly carried (previous distributions remain).

**Final Answer (Row n-1):**
- For each entry `L`, `ways =` ways with `s ≤ 0` (ending runs at row `n-1`).
- Add `ways` if `suffix_row[n-1][L] ≤ 0` (horizontal to column `m-1` legal).

**Special Case n=1:**
- Directly check if `suffix_row[0][0] ≤ 0`.

This ensures all horizontal and vertical segment conditions are satisfied, with time O(n m log n) via segment trees and prefix sums.

    s = Fraction(1)
    num_abs = abs(s.numerator)
    if len(str(num_abs)) >= digits:
        return [int(s.numerator), int(s.denominator)]

        s += term
        num_abs = abs(s.numerator)
        if len(str(num_abs)) >= digits:
            break
        n += 1
    return [int(s.numerator), int(s.denominator)]

import re
import sys

def differentiate(poly, x_val):
    poly = poly.replace(' ', '')
    parts = re.split(r'([+-])', poly)
    terms = []
    if parts and parts[0] == '':
        terms.append((parts[1], parts[2]))
        i = 3
    else:
        terms.append(('+', parts[0]))
        i = 1
    while i < len(parts):
        terms.append((parts[i], parts[i + 1]))
        i += 2
    der_value = 0
    for sign, t in terms:
        if 'x' not in t:
            continue
        x_pos = t.find('x')
        left = t[:x_pos]
        right = t[x_pos + 1:]
        if right and right.startswith('^'):
            power = int(right[1:])
        else:
            power = 1
        if left == '':
            coeff = 1
        else:
            coeff = int(left)
        if sign == '-':
            coeff *= -1
        contrib = coeff * power * (x_val ** (power - 1))
        der_value += contrib
    return der_value

def longest_palindrome(s):
    if not s:
        return ""
    t = '^#' + '#'.join(s) + '#$'
    P = [0] * len(t)
    C = 0
    R = 0
    max_len = 0
    max_center = 0
    for i in range(1, len(t) - 1):
        if i < R:
            mirror = 2 * C - i
            P[i] = min(R - i, P[mirror])
        while t[i + (P[i] + 1)] == t[i - (P[i] + 1)]:
            P[i] += 1
        if i + P[i] > R:
            C = i
            R = i + P[i]
        if P[i] > max_len:
            max_len = P[i]
            max_center = i
    start = (max_center - max_len) // 2
    return s[start:start + max_len]

    spf = list(range(maxa + 1))
    for i in range(2, int(maxa ** 0.5) + 1):

                if spf[j] == j:
                    spf[j] = i
    return spf
