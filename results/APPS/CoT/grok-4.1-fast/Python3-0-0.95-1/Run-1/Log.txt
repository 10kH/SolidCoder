from collections import Counter
import sys

def getAllPrimeFactors(n: int) -> list[int]:
    if n <= 0:
        return []
    factors = []
    while n % 2 == 0:
        factors.append(2)
        n //= 2
    i = 3
    while i * i <= n:
        while n % i == 0:
            factors.append(i)
            n //= i
        i += 2
    if n > 1:
        factors.append(n)
    if not factors:
        return [1]
    return factors

def getUniquePrimeFactorsWithCount(n: int) -> list[list[int]]:
    if n <= 0:
        return [[], []]
    factors = getAllPrimeFactors(n)
    c = Counter(factors)
    primes = sorted(c)
    counts = [c[p] for p in primes]
    return [primes, counts]

def getUniquePrimeFactorsWithProducts(n: int) -> list[int]:
    if n <= 0:
        return []
    factors = getAllPrimeFactors(n)
    c = Counter(factors)
    res = [p ** c[p] for p in sorted(c)]
    if not res:
        return [1]
    return res

# Main execution
try:
    s = sys.stdin.read().strip()
    n = int(s)
    print(getAllPrimeFactors(n))
except ValueError:
    print('[]')

def is_madhav_array(a):
    n = len(a)
    if n <= 1:
        return False
    # Check if n is a triangular number
    discriminant = 1 + 8 * n
    sqrt_d = int(discriminant ** 0.5)
    if sqrt_d * sqrt_d != discriminant:
        return False
    k = (sqrt_d - 1) // 2
    if k * (k + 1) // 2 != n:
        return False
    # Verify sums
    s = a[0]
    pos = 1
    for i in range(2, k + 1):
        current_sum = sum(a[pos:pos + i])
        if current_sum != s:
            return False
        pos += i
    return True

    # Step 1: Round the price P at the tens place
    rounded = ((p + 5) // 10) * 10

    elif r == 0:
        T -= 1
    elif r == -1:

    # Ensure non-negative

def merge_and_count(left, right):
    i = j = 0
    merged = []
    inv = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            merged.append(left[i])
            i += 1
        else:
            merged.append(right[j])
            j += 1
            inv += len(left) - i
    merged.extend(left[i:])
    merged.extend(right[j:])
    return inv, merged

def count_inversions(arr):
